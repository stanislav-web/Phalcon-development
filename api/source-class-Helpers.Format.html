 92: 
 93:         $s = "";
 94:

 96:             if ($quot = intval($secs / $divisor)) {
 97:                 $s .= "$quot $name";
 98:                 $s .= (abs($quot) > 1 ? "s" : "") . " ";
 99:                 $secs -= $quot * $divisor;

101:         }
